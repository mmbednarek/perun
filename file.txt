
fn fib_impl(init_a: i32, init_b: i32, n: i32): i32 {
  var a = init_a;
  var b = init_b;
  var tmp = 0;
  var index = 0;

  while index < n {
    tmp = a + b;
    a = b;
    b = tmp;
    index = index + 1;
  }

  return b;
}

fn fib(n: i32): i32 {
  var value = 5;
  var value_ptr = &value;
  *value_ptr = n;

  // print("hello");

  return fib_impl(1, 1, value);
}
